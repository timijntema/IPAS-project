115  // configure a pin as ADC
116  int __attribute__((weak)) configure_as_adc( int port, int pin ){
117 
118  // enable IOCON & A/D clock
119  LPC_SYSCON->SYSAHBCLKCTRL |= ( 0x01 << 16 ) | ( 0x01 << 13 );
120 
121  // enable A/D power
122  LPC_SYSCON->PDRUNCFG &= ~( 1 << 4 );
123 
124  switch( ( port << 8 ) | pin ){
125  case 0x000B: pin_ad_conf( &LPC_IOCON->R_PIO0_11, 0x02 ); return 0;
126 
127  case 0x0100: pin_ad_conf( &LPC_IOCON->R_PIO1_0, 0x02 ); return 1;
128  case 0x0101: pin_ad_conf( &LPC_IOCON->R_PIO1_1, 0x02 ); return 2;
129  case 0x0102: pin_ad_conf( &LPC_IOCON->R_PIO1_2, 0x02 ); return 3;
130  case 0x0103: pin_ad_conf( &LPC_IOCON->SWDIO_PIO1_3, 0x02 ); return 4;

282  adc_value_type get() override {
283 
284  // configure the A/D for the pin
285  LPC_ADC->CR = ( 0x01 << channel ) | ( 12 << 8 );
286 
287  // start the conversion
288  LPC_ADC->CR |= ( 0x01 << 24 );
289 
290  // wait for the conversion to complete
291  while( ( LPC_ADC->GDR & ( 1 << 31 )) == 0 );
292 
293  hwlib::wait_ms( 10 );
294 
295  // return the A/D result
296  return 3 | (( LPC_ADC->GDR >> 6 ) & 0x3FF);

311 void HWLIB_WEAK wait_us( int_fast32_t n ){
312  // for the default clock frequency of 12 MHz:
313  // the loop uses 12 cycles
314  __asm volatile(
315  " mov r0, %[reg] \t\n"
316  " b 1f \t\n"
317  " .align 4 \t\n"
318  "1: sub r0, #1 \t\n"
319  " b 2f \t\n"
320  "2: b 3f \t\n"
321  "3: nop \t\n"
322  " nop \t\n"
323  " bgt 1b \t\n"
324  :: [reg] "r" (n) : "r0"
325  );
326 }
