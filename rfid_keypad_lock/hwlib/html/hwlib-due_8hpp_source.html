189 Pio & __attribute__((weak)) port_registers( int port ){
190  // a bit of a cludge to put this here:
191  // enable the clock to all GPIO ports
192  PMC->PMC_PCER0 = ( 0x3F << 11 );
193 
194  switch( port ){
195  case 0 : return *PIOA;

198  case 3 : return *PIOD;
199  default : break;
200  }

202  // doesn't return
203 }

408 void HWLIB_WEAK wait_us( int_fast32_t n ){
409 
410  // use the 12 MHz internal RC clock
411  PMC->CKGR_MOR = 0x00370028;
412 
413  // This code is for a clock frequency of 12 MHz:
414  // the loop uses exactly 12 cycles on an M3
415  // (1 cycle branch penalty due to branch target forwarding).
416  // Note: use subs, sub on an M3 is the wide instruction
417  // that doesn't set the status flags.
418  __asm volatile(
419  " mov r0, %[reg] \t\n"
420  " b 1f \t\n"
421  " .align 4 \t\n"
422  "1: subs r0, #1 \t\n"
423  " b 2f \t\n"
424  "2: b 3f \t\n"
425  "3: b 4f \t\n"
426  "4: b 5f \t\n"
427  "5: nop \t\n"
428  " bgt 1b \t\n"
429  :: [reg] "r" (n) : "r0"
430  );
431 }
