412 void HWLIB_WEAK wait_us( int_fast32_t n ){
413  // first int parameter is passd in r24/r25
414  __asm volatile( // clocks
415  "1: cp r1, r24 \t\n" // 1
416  " cpc r1, r25 \t\n" // 1
417  " brge 3f \t\n" // 1
418  " rcall 3f \t\n" // 7
419  " rjmp 2f \t\n" // 2
420  "2: sbiw r24, 0x01 \t\n" // 2
421  " rjmp 1b \t\n" // 2
422  "3: \t\n" // 16 total
423  );
424 }
