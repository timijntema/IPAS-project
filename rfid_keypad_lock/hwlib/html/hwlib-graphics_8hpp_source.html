310 class line : public drawable {
311 private:
312  location end;
313  color fg;
314 
315  static void swap( int & a, int & b ){
316  unsigned int t;
317  t = a;
318  a = b;
319  b = t;
320  }
321 
322  static int abs( int x ){
323  return x >= 0 ? x : -x;
324  }
325 
326 public:
327  line( location start, location end, color fg = black )
328  : drawable{ start }, end{ end }, fg{ fg }
329  {}
330 
331 
332  void draw( window & w ) override {
333  int x0 = start.x;
334  int y0 = start.y;
335  int x1 = end.x;
336  int y1 = end.y;
337 
338  // http://en.wikipedia.org/wiki/Bresenham%27s_line_algorithm
339  // http://homepages.enterprise.net/murphy/thickline/index.html
340 
341  int Dx = x1 - x0;
342  int Dy = y1 - y0;
343 
344  int steep = (abs(Dy) >= abs(Dx));
345 
346  if( steep ){
347  swap( x0, y0 );
348  swap( x1, y1 );
349 
350  // recompute Dx, Dy after swap
351  Dx = x1 - x0;
352  Dy = y1 - y0;
353  }
354 
355  int xstep = 1;
356  if( Dx < 0 ){
357  xstep = -1;
358  Dx = -Dx;
359  }
360 
361  int ystep = 1;
362  if( Dy < 0 ){
363  ystep = -1;
364  Dy = -Dy;
365  }
366  int TwoDy = 2*Dy;
367  int TwoDyTwoDx = TwoDy - 2*Dx; // 2*Dy - 2*Dx
368  int E = TwoDy - Dx; //2*Dy - Dx
369  int y = y0;
370  int xDraw, yDraw, x;
371  for( x = x0; x != x1; x += xstep ){
372  if (steep) {
373  xDraw = y;
374  yDraw = x;
375  } else {
376  xDraw = x;
377  yDraw = y;
378  }
379 
380  w.write( location{ xDraw, yDraw }, fg );
381 
382  if( E > 0 ){
383  E += TwoDyTwoDx; //E += 2*Dy - 2*Dx;
384  y = y + ystep;
385  } else {
386  E += TwoDy; //E += 2*Dy;
387  }
388  }
389  }

394 private:

401  : drawable{ start }, radius{ radius }, fg{ fg }, bg{ bg }
402  {}
403 
404  void draw( window & w ) override {
405 
406  // don't draw anything when the size would be 0
407  if( radius < 1 ){
408  return;
409  }
410 
411  // http://en.wikipedia.org/wiki/Midpoint_circle_algorithm
412 
413  int fx = 1 - radius;
414  int ddFx = 1;
415  int ddFy = -2 * radius;
416  int x = 0;
417  int y = radius;
418 
419  // top and bottom
420  w.write( start + location( 0, + radius ), fg );
421  w.write( start + location( 0, - radius ), fg );
422 
423  // left and right
424  w.write( start + location( + radius, 0 ), fg );
425  w.write( start + location( - radius, 0 ), fg );
426 
427  if( bg != transparent ){
428 
429  // top and bottom
430  w.write( start + location( 0, + radius ), fg );
431  w.write( start + location( 0, - radius ), fg );
432 
433  // left and right
434  line(
435  start - location( radius, 0 ),
436  start + location( radius, 0 ),
437  fg
438  ).draw( w );
439  }
440 
441  while( x < y ){
442 
443  // calculate next outer circle point
444  if( fx >= 0 ){
445  y--;
446  ddFy += 2;
447  fx += ddFy;
448  }
449  x++;
450  ddFx += 2;
451  fx += ddFx;
452 
453  w.write( start + location( + x, + y ), fg );
454  w.write( start + location( - x, + y ), fg );
455  w.write( start + location( + x, - y ), fg );
456  w.write( start + location( - x, - y ), fg );
457  w.write( start + location( + y, + x ), fg );
458  w.write( start + location( - y, + x ), fg );
459  w.write( start + location( + y, - x ), fg );
460  w.write( start + location( - y, - x ), fg );
461 
462  if( bg != transparent ){
463  line( start + location( -x, y ), start + location( x, y ), bg ).draw( w );
464  line( start + location( -x, -y ), start + location( x, -y ), bg ).draw( w );
465  line( start + location( -y, x ), start + location( y, x ), bg ).draw( w );
466  line( start + location( -y, -x ), start + location( y, -x ), bg ).draw( w );
467  }
